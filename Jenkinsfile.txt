node {

	stage('Git_clone') {
		echo 'cloning....'
        sh 'git clone https://github.com/eswar629/pipeline.git'
	}

	stage('Terraform') {
		echo 'Terraform init'
        sh 'terraform init'
        sh 'terraform plan --var-file -var test.tfvars'
        
	}

	stage('Deploy') {
		echo 'Deploying....'
        sh 'terraform apply --var-file test.tfvars'
	}
}
